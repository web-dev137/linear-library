102 void LU<T, MatrixType>::decomposition() {
103 if (matrix.getRows() != matrix.getCols())
104 throw std::runtime_error("Matrix must be square for LU decomposition");
105 int n = matrix.getRows();
106 initP();
107 for (int k = 0; k < n; ++k) {
108 int pivot = pivoting(k);
109 if(pivot != k) {
110 for (int j = 0; j < n; j++) {
111 std::swap(matrix(k,j), matrix(pivot,j));
112 }

114 std::swap(P[k], P[pivot]);

118 throw std::runtime_error("Matrix is singular or nearly singular at pivot " + std::to_string(k));
